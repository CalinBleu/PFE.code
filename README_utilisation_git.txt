#Règles git

#Pour mettre à jour les ajouts 
git pull

#Pour ajouter un document 
git add <document à ajouter>

#Pour commit le document ajouté
git commit -m "<message>"

#Les messages doivent être conçus de cette façon
<Mot majeur>- <Info sur la modification du doc que vous avez modifié>

#Mots majeurs
CREATE : si c'est un nouveau document que vous ajoutez
MAJ : Si vous modifiez un doc existant
BUG : Défault à corriger

#Pour mettre les documents sur le git global
git push
